class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        #动态规划
        # l = 0
        # dp = [1 for i in nums]
        # for i in range(len(nums)):
        #     for j in range(i):
        #         if nums[j]<nums[i]:
        #             dp[i] = max(dp[i], dp[j]+1)
        #     l = max(l, dp[i])
        # return l

        ## 二分查找
        if len(nums) <=1: return len(nums)
        dp = [nums[0]]
        for i in range(1, len(nums)):
            left = 0
            right = len(dp)-1
            while left <= right:
                mid = int((left+right)/2)
                if dp[mid] < nums[i]:
                    left = mid+1
                else:
                    right = mid-1
            if left == len(dp):
                dp.append(nums[i])
            else:
                dp[left] =nums[i]
        return len(dp)
