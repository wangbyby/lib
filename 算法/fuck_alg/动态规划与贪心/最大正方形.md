
# 最大正方形

```python

class Solution:
    def maximalSquare(self, matrix: List[List[str]]) -> int:
        if len(matrix) == 0:return 0
        m = len(matrix)
        n = len(matrix[0])
        dp = [[0 for j in range(1+n)] for i in range(1+m)]
        mS = 0
        for ii in range(m):
            for ji in range(n):
                val = matrix[ii][ji]
                i = ii+1
                j = ji+1
                if val == '1':
                    ms = min(dp[i][j-1],dp[i-1][j],dp[i-1][j-1])
                    # dp[i][j] = 1+ms+2*int(pow(ms,0.5))
                    dp[i][j] = ms+1 #边长
                mS = max(mS,dp[i][j])
        return mS*mS
```
